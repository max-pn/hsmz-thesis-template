\RequirePackage{xparse}
\RequirePackage{caption}
\RequirePackage{longtable}
\RequirePackage{booktabs}
\RequirePackage{array}
\RequirePackage{ragged2e}
\RequirePackage{setspace}

\NewDocumentCommand{\inserttable}{ m +m m m o }{%
	\begingroup

	\captionsetup{%
		position=top,
		format=plain,
		justification=raggedright,
		singlelinecheck=false,
		labelfont=bf,
		textfont=it,
		labelsep=newline,
		font=apa-caption-font
	}%

	\setlength\LTleft{0pt}%
	\setlength\LTright{0pt}%
	\setlength\LTpost{0pt}%

	\begin{longtable}{#1}
		\caption{#3}#4 \\
		#2
	\end{longtable}

	\IfValueT{#5}{%
		\captionsetup{%
			position=bottom,
			format=plain,
			justification=raggedright,
			singlelinecheck=false,
			font=normalfont
		}%
		\captionof*{table}{\textit{\tableAnnotation.}\ \normalfont #5}%
		\vspace{0.75\baselineskip}%
	}%

	\endgroup
}
