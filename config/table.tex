\RequirePackage{xparse}
\RequirePackage{caption}
\RequirePackage{adjustbox}

\NewDocumentCommand{\inserttable}{ O{\linewidth} +m m m o o }{%
	\begin{table}[H]
		\raggedright

		\captionsetup{%
			position=top,
			format=plain,
			justification=raggedright,
			singlelinecheck=false,
			labelfont=bf,
			textfont=it,
			labelsep=newline,
			font=apa-caption-font
		}%

		\IfValueTF{#6}
		{\caption[#6]{#3}}
		{\caption{#3}}

		#4 % write \label{} in the \inserttable{} call not here, so lsp recognizes it and suggests it for usage

		\vspace{0.5\baselineskip}

		\begin{adjustbox}{max width=#1}
			#2
		\end{adjustbox}\par

		\IfValueT{#5}{%
			\captionsetup{%
				position=bottom,
				format=plain,
				justification=raggedright,
				singlelinecheck=false,
				font=normalfont
			}%
			\caption*{\textit{\tableAnnotation.}\ \normalfont #5}
		}%
	\end{table}%
}
