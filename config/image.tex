\RequirePackage{graphicx}
\RequirePackage{xparse}
\RequirePackage{caption}


\NewDocumentCommand{\insertimage}{ O{\linewidth} m m m o o}{%
	\begin{figure}[H]
		\raggedright

		\captionsetup{%
			position=top,
			format=plain,
			justification=raggedright,
			singlelinecheck=false,
			labelfont=bf,
			textfont=it,
			labelsep=newline,
			font=apa-caption-font
		}%

		\IfValueTF{#6}
		{\caption[#6]{#3}}
		{\caption{#3}}

		#4 % write \label{} in the \inserttable{} call not here, so lsp recognizes it and suggests it for usage

		\includegraphics[width=#1]{#2}\par

		\IfValueT{#5}{%
			\captionsetup{%
				position=bottom,
				format=plain,
				justification=raggedright,
				singlelinecheck=false,
				font=normalfont
			}%
			\caption*{\textit{\figureAnnotation.}\ \normalfont #5}
		}

	\end{figure}%
}
